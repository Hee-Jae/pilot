import java.io.*;

class Main implements Runnable {
	static String ReadLn(int maxLength) {
		byte line[] = new byte[maxLength];
		int length = 0;
		int input = -1;
		try {
			while (length < maxLength) {
				input = System.in.read();
				if ((input < 0) || (input == '\n'))
					break;
				line[length++] += input;
			}

			if ((input < 0) && (length == 0))
				return null;
			return new String(line, 0, length);
		} catch (IOException e) {
			return null;
		}
	}

	public static void main(String args[]) {
		Main myWork = new Main();
		myWork.run();
	}

	public void run() {
		new myStuff().run();
	}
}

class myStuff implements Runnable {
	public void run() {
		String inputText;
		while ((inputText = Main.ReadLn(10000000)) != null) {
			inputText = inputText.trim();
			if (inputText.isEmpty())
				continue;

			String[] t = inputText.split("\\s+");

			int i = Integer.parseInt(t[0].trim());
			int j = Integer.parseInt(t[1].trim());

			System.out.printf("%d %d %d\n", i, j, this
					.maxLengthOfThreeNPlusOne(i, j));
		}
	}

	private int threeNplusOne(int n) {
		int cnt = 1;
		while (n != 1) {
			if (n % 2 == 0) {
				n = n / 2;
				cnt++;
			} else {
				n = 3 * n + 1;
				cnt++;
			}
		}
		return cnt;
	}

	private int maxLengthOfThreeNPlusOne(int n, int m) {

		int temp = 0;
		if (n > m) {
			temp = n;
			n = m;
			m = temp;
		}

		int max = 0;
		for (int i = n; i <= m; i++) {
			int lTemp = threeNplusOne(i);
			if (max < lTemp) {
				max = lTemp;
			}
		}

		return max;
	}
}
